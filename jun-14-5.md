class Solution:ds
    def pickGifts(self, gifts: List[int], k: int) -> int:
        while k > 0:
            k -= 1
            m = float('-inf')
            mi = None
            for i, g in enumerate(gifts):
                if g > m:
                    m = g
                    mi = i
            gifts[mi] = int(m ** 0.5)
        return sum(gifts)
